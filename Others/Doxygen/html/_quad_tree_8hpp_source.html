142 template <class T>
143 void QuadTreeNode<T>::insert(Point<T> new_point){
145  if (!hasData()){
146  this->point = new Point<T>[1];
147  point[0] = new_point;
148  number_of_points=1;
149  }
151  else if (hasData() && level != MAX_LEVEL){
153  split();
155  size_t i=3;
156  if (point->x <= x+width/2 && point->y > y+height/2) i=0;
157  else if (point->x > x+width/2 && point->y >= y+height/2) i=1;
158  else if (point->x > x+width/2 && point->y <= y+height/2) i=2;
159  children[i]->insert(*point);
160  delete[] point;
161  point=NULL;
162  number_of_points=0;
164  i=3;
165  if (new_point.x <= x+width/2 && new_point.y > y+height/2) i=0;
166  else if (new_point.x > x+width/2 && new_point.y >= y+height/2) i=1;
167  else if (new_point.x > x+width/2 && new_point.y <= y+height/2) i=2;
168  children[i]->insert(new_point);
169  }
171  else{
173  Point<T> *temp = new Point<T>[number_of_points+1];
174  size_t i;
176  for (i=0; i<number_of_points; ++i)
177  temp[i]=point[i];
179  temp[number_of_points]=new_point;
181  ++number_of_points;
183  delete[] point;
184  point = temp;
185  }
186 }
