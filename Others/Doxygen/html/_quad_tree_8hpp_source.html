138 void QuadTreeNode<T>::insert(Point<T> new_point){
140  if (!hasData()){
141  this->point = new Point<T>[1];
142  point[0] = new_point;
143  number_of_points=1;
144  }
146  else if (hasData() && level != MAX_LEVEL){
148  split();
150  size_t i=3;
151  if (point->x <= x+width/2 && point->y > y+height/2) i=0;
152  else if (point->x > x+width/2 && point->y >= y+height/2) i=1;
153  else if (point->x > x+width/2 && point->y <= y+height/2) i=2;
154  children[i]->insert(*point);
155  delete[] point;
156  point=NULL;
157  number_of_points=0;
159  i=3;
160  if (new_point.x <= x+width/2 && new_point.y > y+height/2) i=0;
161  else if (new_point.x > x+width/2 && new_point.y >= y+height/2) i=1;
162  else if (new_point.x > x+width/2 && new_point.y <= y+height/2) i=2;
163  children[i]->insert(new_point);
164  }
166  else{
168  Point<T> *temp = new Point<T>[number_of_points+1];
169  size_t i;
171  for (i=0; i<number_of_points; ++i)
172  temp[i]=point[i];
174  temp[number_of_points]=new_point;
176  ++number_of_points;

179  point = temp;
